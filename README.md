# Для правильного приховування заголовків використовуємо паттерн – visually-hidden.
// HTML
<h2 class="visually-hidden">Title<h2>

// CSS
.visually-hidden {
  position: absolute;
  width: 1px;
  height: 1px;
  margin: -1px;
  border: 0;
  padding: 0;
  white-space: nowrap;
  clip-path: inset(100%);
  clip: rect(0 0 0 0);
  overflow: hidden;
}
